LeetCode 977. 有序数组的平方

class Solution {
    public int[] sortedSquares(int[] nums) {
        int n = nums.length;
        int[] ans = new int[n];
        int pos = n - 1;
        for(int i = 0,j = n-1;i <= j;pos--){
            if(nums[i] * nums[i] > nums[j] * nums[j]){
                ans[pos] = nums[i] * nums[i];
                i ++;
            }else{
                ans[pos] = nums[j] * nums[j];
                j --;
            }
        }
        return ans;
    }
}

方法一：使用额外数组

    public void rotate(int[] nums, int k) {
        int len = nums.length;
        int[] ans = new int[len];
        for(int i = 0; i < len ;i++){
            ans[(i + k) % len] = nums[i];
        }
        //复制数组
        System.arraycopy(ans,0,nums,0,len);
    }

    public void rotate(int[] nums, int k) {
        int n = nums.length;
        k %= n;
        reverse(nums,0,n-1);
        reverse(nums,0,k-1);
        reverse(nums,k,n-1);
    }

    public void reverse(int[] nums, int start,int end){
        while(start < end){
            int temp = nums[start];
            nums[start] = nums[end];
            nums[end] = temp;
            start++;
            end--;
        }
    }

LeetCode 283. 移动零

方法一：覆盖前面的值，后续补零

    public void moveZeroes(int[] nums) {
        int j = 0;
        for(int i = 0;i < nums.length;i++){
            if(nums[i] != 0){
                nums[j++] = nums[i];
            }
        }
        for(int i = j;i < nums.length;i++){
            nums[i] = 0;
        }
    }

方法二：双指针

    public void moveZeroes(int[] nums) {
        //left指向已处理好的序列尾部
        //right指向待处理的序列头部
        int n = nums.length,left = 0,right = 0;
        while(right < n){
            if(nums[right] != 0){
                swap(nums,left,right);
                left++;
            }
            right++;
        }
    }

    public void swap(int[] nums,int i,int j){
        int temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }

LeetCode 167. 两数之和 II - 输入有序数组

    public int[] twoSum(int[] numbers, int target) {
        if(numbers.length == 2) return new int[]{1,2};
        int n = numbers.length;
        int[] ans = new int[2];
        int left = 0;
        int right = n-1;
        while(left < right){
            if(numbers[left] + numbers[right] == target){
                ans[0] = left+1;
                ans[1] = right+1;
                return ans;
            }else if(numbers[left] + numbers[right] > target){
                right --;
            }else{
                left ++;
            }
        }
        return ans;
    }

LeetCode 344. 反转字符串

    public void reverseString(char[] s) {
        int left = 0;
        int right = s.length - 1;
        while(left < right){
            swap(s,left++,right--);
        }
    }

    public void swap(char[] s,int i,int j){
        char temp = s[i];
        s[i] = s[j];
        s[j] = temp;
    }

LeetCode 876. 链表的中间结点

快慢指针

    public ListNode middleNode(ListNode head) {
        if(head == null) return null;
        ListNode fast = head,slow = head;
        //当链表长度是偶数时，需要判断fast.next,如果fast.next == null 说明fast已经走到链表尾端
        while(fast != null && fast.next != null){
            fast = fast.next.next;
            slow = slow.next;
        }
        return slow;
    }

LeetCode 19. 删除链表的倒数第 N 个结点

快慢指针

  1. 将快指针start、慢指针end指向预先节点dummy
  2. start先走n个节点，然后start，end一起走
  3. 当start走到链表最后一个节点时，end就到了链表倒数第 n + 1 个节点。
    public ListNode removeNthFromEnd(ListNode head, int n) {
        if(head.next == null) return null;
        //防止head节点被删除，设置预先节点指向head
        ListNode dummy = new ListNode();
        dummy.next = head;
        ListNode start = dummy;
        ListNode end = dummy;
        for(int i = 0;i < n;i++){
            start = start.next;
        }
        while(start.next != null){
            start = start.next;
            end = end.next;
        }
        end.next = end.next.next;
        return dummy.next;
    }

LeetCode 11. 盛最多水的容器

双指针

  1. 初始化： 双指针 i , j 分列水槽左右两端；
  2. 循环收窄： 直至双指针相遇时跳出；
  3. 更新面积最大值 res ；
  4. 选定两板高度中的短板，向中间收窄一格；
  5. 返回值： 返回面积最大值 res 即可；
    public int maxArea(int[] height) {
        int res = 0;        //结果
        int S = 0;            //每次循环时的面积
        int i = 0,j = height.length - 1;
        while(i < j){
            S = Math.min(height[i],height[j]) * (j - i);
            res = Math.max(res,S);
            //选择移动的指针
            if(height[i] < height[j]){
                i++;
            }else{
                j--;
            }
        }
        return res;
    }
