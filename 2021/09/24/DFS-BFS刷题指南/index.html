LeetCode 617. 合并二叉树

「方法一」DFS

class Solution {
    public TreeNode mergeTrees(TreeNode root1, TreeNode root2) {
        if(root1 == null) return root2;
        if(root2 == null) return root1;
        root1.val = root1.val+root2.val;
        root1.left = mergeTrees(root1.left,root2.left);
        root1.right = mergeTrees(root1.right,root2.right);
        return root1;
    }
}

LeetCode 116. 填充每个节点的下一个右侧节点指针

「方法一」DFS

class Solution {
    public Node connect(Node root) {
        if(root == null) return null;
        dfs(root);
        return root;
    }

    public void dfs(Node node){
        if(node.left != null){
            node.left.next = node.right;
            if(node.next != null){
                node.right.next = node.next.left;
            }
        }
        if(node.left != null) dfs(node.left);
        if(node.right != null) dfs(node.right);
    }
}

「方法二」BFS

class Solution {
    public Node connect(Node root) {
        if(root == null) return null;
        Queue<Node> queue = new LinkedList<>();
        queue.offer(root);
        while(!queue.isEmpty()){
            int n = queue.size();

            for(int i = 0;i < n;i++){
                Node node = queue.poll();
                if(i < n - 1){
                    node.next = queue.peek();
                }
                if(node.left != null){
                    queue.offer(node.left);
                }
                if(node.right != null){
                    queue.offer(node.right);
                }
            }
            
        }
        return root;
    }
}

LeetCode 542. 01 矩阵

「方法一」BFS

class Solution {
    public int[][] updateMatrix(int[][] mat) {

        int[] dx = {1,0,0,-1};
        int[] dy = {0,1,-1,0};
        //模拟队列
        Queue<int[]> queue = new LinkedList<>();

        int rn = mat.length;
        int cn = mat[0].length;

        //初始化队列、二维数组
        for(int i = 0; i < rn; i++){
            for(int j = 0; j < cn; j++){
                if(mat[i][j] == 0){
                    queue.offer(new int[]{i,j});
                }else{
                    mat[i][j] = -1;
                }
            }
        }

        while(!queue.isEmpty()){
            int[] cell = queue.poll();
            int x = cell[0];
            int y = cell[1];
            for(int i = 0; i < 4; i++){
                int newX = x + dx[i];
                int newY = y + dy[i];
                //当遇到未处理的位置
                if(newX >= 0 && newX < rn && newY >= 0 && newY < cn && mat[newX][newY] == -1){
                    //设置位置信息并且将该位置加入队列
                    queue.offer(new int[]{newX,newY});
                    mat[newX][newY] = mat[x][y] + 1;
                }
            }
        }
        return mat;

    }
}

LeetCode 994. 腐烂的橘子

「方法一」BFS

class Solution {
    public int orangesRotting(int[][] grid) {
        int[] dx = {1,0,0,-1};
        int[] dy = {0,1,-1,0};

        int rn = grid.length;
        int cn = grid[0].length;

        Queue<Integer> queue = new LinkedList<>();
        Map<Integer,Integer> map = new HashMap<>();

        for(int i = 0; i < rn; i++){
            for(int j = 0; j < cn; j++){
                if(grid[i][j] == 2){
                    int rc = i * cn + j;
                    queue.offer(rc);
                    map.put(rc,0);
                }
            }
        }

        int ans = 0;
        while(!queue.isEmpty()){
            int rc = queue.poll();
            int x = rc / cn;
            int y = rc % cn;
            for(int i = 0; i < 4; i++){
                int newX = x + dx[i];
                int newY = y + dy[i];
                if(newX >= 0 && newX < rn && newY >= 0 && newY < cn && grid[newX][newY] == 1 ){
                    int newRc = newX * cn + newY;
                    queue.offer(newRc);
                    grid[newX][newY] = 2;
                    map.put(newRc ,map.get(rc) + 1);      
                    ans = map.get(newRc);         
                }
            }
        }

        for(int[] nums:grid){
            for(int num: nums){
                if(num == 1)   return -1;
            }
        }

        return ans;
    }
}
