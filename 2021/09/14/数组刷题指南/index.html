LeetCode 217. 存在重复元素

方法一：排序后查找

    public boolean containsDuplicate(int[] nums) {
        Arrays.sort(nums);
        for(int i = 0;i < nums.length - 1;i++){
            if(nums[i] == nums[i+1]){
                return true;
            }
        }
        return false;
    }

方法二：HashMap方法

    public boolean containsDuplicate(int[] nums) {
        Map<Integer,Integer> map = new HashMap<>();
        for(int num:nums){
            if(map.get(num) == null){
                map.put(num,1);
            }else{
                return true;
            }
        }
        return false;
    }

LeetCode 53. 最大子序和

方法一：动态规划

    //原始递归
    public int maxSubArray(int[] nums) {
        int ans = Integer.MIN_VALUE;
        for(int i = 0;i < nums.length;i++){
            ans = Math.max(ans,dp(nums,i));
        }
        return ans;
    }
    //dp(i)表示以第i个数结尾的「连续子数组的最大和」
    public int dp(int[] nums,int i){
        if(i == 0) return nums[0];
        return Math.max(dp(nums,i - 1) + nums[i] , nums[i]);
    }

    //改良版
    public int maxSubArray(int[] nums) {
        int pre = 0;
        int ans = Integer.MIN_VALUE;
        for(int num:nums){
            pre = Math.max(pre + num , num );
            ans = Math.max(pre,ans);
        }
        return ans;
    }

LeetCode 1. 两数之和

使用HashMap存储数据查找第二个数

    public int[] twoSum(int[] nums, int target) {
        Map<Integer,Integer> map = new HashMap<>();
        for(int i = 0;i < nums.length;i++){
            if(map.containsKey(target - nums[i])){
                return new int[]{map.get(target - nums[i]),i};
            }
            map.put(nums[i],i);
        }
        return new int[]{0,1};
    }

LeetCode 88. 合并两个有序数组

使用双指针指向两个数组，开辟一个新数组，将数据存到新数组中。

    public void merge(int[] nums1, int m, int[] nums2, int n) {
        int[] sorted = new int[m + n];
        int p1 = 0,p2 = 0;
        int cur;
        while(p1 < m || p2 < n){
            if(p1 == m){
                cur = nums2[p2++];
            }else if(p2 == n){
                cur = nums1[p1++];
            }else if(nums1[p1] < nums2[p2]){
                cur = nums1[p1++];
            }else{
                cur = nums2[p2++];
            }
            sorted[p1 + p2 - 1] = cur;
        }
        System.arraycopy(sorted,0,nums1,0,m+n);
    }

LeetCode 73. 矩阵置零

使用matrix数组的第0行、第0列存储二维数组中的0的标志信息，但是在这之前要先判断第0行或第0列是否存在0，如果存在后续还需处理第0行、第0列。

时间复杂度：O(nm)

空间复杂度：O(1)

class Solution {
    public void setZeroes(int[][] matrix) {
        //第0行或第0列是否存在0；
        Boolean flag_row0 = false;
        Boolean flag_col0 = false;
        int row = matrix.length;
        int col = matrix[0].length;
        //先判断第0行或第0列是否存在0
        for(int i = 0;i < row;i++){
            if(matrix[i][0] == 0){
                flag_col0 = true;
            }
        }
        for(int j = 0;j < col;j++){
            if(matrix[0][j] == 0){
                flag_row0 = true;
            }
        }
        //使用第0行，第0列标记
        for(int i = 1; i < row;i++){
            for(int j = 1; j < col;j++){
                if(matrix[i][j] == 0){
                    matrix[i][0] = matrix[0][j] = 0;
                }
            }
        }
        //赋值0；
        for(int i = 1; i < row;i++){
            for(int j = 1; j < col;j++){
                if(matrix[i][0] == 0 || matrix[0][j] == 0){
                    matrix[i][j] = 0;
                }
            }
        }
        if(flag_row0){
            for(int j = 0;j < col;j++){
                matrix[0][j] = 0;
            }
        }
        if(flag_col0){
            for(int i = 0;i < row;i++){
                matrix[i][0] = 0;
            }
        }
    }
}

LeetCode 121. 买卖股票的最佳时机

方法一：和LeetCode 53. 最大子序和有点像。

    public int maxProfit(int[] prices) {
        int maxAns = 0;
        //f(i)表示第i天买出可获得的最大利润
        int fi = 0;
        for(int i = 1;i < prices.length;i++){
            if(i == 1){
                fi = Math.max(0,prices[i] - prices[i-1]);
            }else{
                fi = Math.max(fi+prices[i] - prices[i-1],prices[i] - prices[i-1]);
            }
            maxAns = Math.max(fi,maxAns);
        }
        return maxAns;
    }

方法二：前i天的最大收益 = max{前i-1天的最大收益，第i天的价格-前i-1天中的最小价格}

  1. 记录【今天之前买入的最小值】
  2. 计算【今天之前最小值买入，今天卖出的获利】，也即【今天卖出的最大获利】
  3. 比较【每天的最大获利】，取最大值即可
    public int maxProfit(int[] prices) {
        int maxAns = 0;
        //记录【今天之前买入的最小值】
        int min = prices[0];
        for(int i = 1; i < prices.length;i++){
            maxAns = Math.max(maxAns,prices[i] - min);
            min = Math.min(min,prices[i]);
        }
        return maxAns;
    }

LeetCode 350. 两个数组的交集 II

方法一：HashMap存储数组

    public int[] intersect(int[] nums1, int[] nums2) {
        Map<Integer,Integer> map = new HashMap<>();
        for(int num1:nums1){
            if(!map.containsKey(num1)){
                map.put(num1,1);
            }else{
                map.put(num1,map.get(num1) + 1);
            }
        }
        List<Integer> ansList = new ArrayList<>();
        for(int num2:nums2){
            if(map.containsKey(num2) && map.get(num2) > 0){
                ansList.add(num2);
                map.put(num2,map.get(num2) - 1);
            }
        }
        int[] arr = new int[ansList.size()];
        for(int i = 0;i < ansList.size();i++){
            arr[i] = ansList.get(i);
        }
        return arr;
    }

方法二：排序+双指针

class Solution {
    public int[] intersect(int[] nums1, int[] nums2) {
        Arrays.sort(nums1);
        Arrays.sort(nums2);
        int p1 = 0,p2 = 0;
        ArrayList<Integer> ansList = new ArrayList<>();
        while(p1 < nums1.length && p2 < nums2.length){
            if(nums1[p1] == nums2[p2]){
                ansList.add(nums1[p1]);
                p1++;
                p2++;
            }else if(nums1[p1] < nums2[p2]){
                p1++;
            }else{
                p2++;
            }
        }
        int[] ans = new int[ansList.size()];
        int i = 0;
        for(int an:ansList){
            ans[i++] = an;
        }
        return ans;
    }
}
